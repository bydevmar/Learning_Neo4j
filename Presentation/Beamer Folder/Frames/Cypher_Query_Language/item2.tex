  CREATE (:Person {name: 'simo', age: 30})

    \begin{block}{Créer une relation entre deux nœuds :}
   MATCH (a:Person), (b:Person)
WHERE a.name = 'simo' AND b.name = 'walid'
CREATE (a)-[:FRIEND]->(b)

  \end{block}
   \begin{block}{Trouver tous les nœuds d'un type spécifique :}
   MATCH (p:Person)
RETURN p
